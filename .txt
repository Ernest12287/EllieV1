## üéØ QUICK CREATION GUIDE:

### For each command, follow this structure:

```javascript
import config from '../config.js';

export default {
  name: 'commandname',
  aliases: ['alias1', 'alias2'],
  description: 'Description! üéØ',
  usage: '.command <args>',
  category: 'Category',
  async execute(sock, message, args) {
    const sender = message.key.remoteJid;
    
    // 1. Check args
    if (args.length < 1) {
      return await sock.sendMessage(sender, { 
        text: `Help message with usage examples`
      }, { quoted: message });
    }

    // 2. Build API URL
    const param = args.join(' ');
    const apiUrl = `https://api.giftedtech.co.ke/api/.../...?apikey=gifted&param=${encodeURIComponent(param)}`;

    try {
      // 3. Send loading message
      await sock.sendMessage(sender, { 
        text: `‚è≥ Processing...`
      }, { quoted: message });

      // 4. Fetch API
      const response = await fetch(apiUrl);
      const data = await response.json();

      // 5. Send result
      if (data.success && data.result) {
        // Format and send response
        await sock.sendMessage(sender, { 
          text: `Result: ${data.result}`
        }, { quoted: message });
      } else {
        await sock.sendMessage(sender, { 
          text: `‚ùå Failed!`
        }, { quoted: message });
      }
    } catch (error) {
      console.error('Error:', error);
      await sock.sendMessage(sender, { 
        text: `‚ùå Error! ${error.message}`
      }, { quoted: message });
    }
  }
};
```
